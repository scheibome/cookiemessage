Init cookiemessage

document.getElementById('cookiemessage').cookiemessage();

Settings

lifetime Lifetime of cookie in days int(days) 365 
document.getElementById('cookiemessage').cookiemessage({
	lifetime: 365
});
button Allows you to select a node or customize the HTML for the "Accept" button. object 
<button id="cookiemessagebutton">Accept</button>
 
document.getElementById('cookiemessage').cookiemessage({
	button: document.getElementById('cookiemessage').getElementsByTagName('button')[0]
});
position Set the position of the message. You can also use 'bottom-left' or 'bottom'. string bottom-right 
document.getElementById('cookiemessage').cookiemessage({
	position: 'bottom-left'
});
palette Allows you to adjust the color of the cookie message. object 
document.getElementById('cookiemessage').cookiemessage({
	palette: {
		message: {
			background: '#000',
			text: '#fff'
		},
		button: {
			background: '#ff0000',
			text: '#fff'
		}
	}
});
pushbody Adds a padding to the body with the height of the message so that the message does not cover the content of the web page. boolean true 
document.getElementById('cookiemessage').cookiemessage({
	pushbody: true
});
